title: Check if a value is base64 encoded
category: Validator
---

**JavaScript version**

```js
const isBase64 = (value) =>
	/^(?:[A-Za-z0-9+/]{4})*(?:[A-Za-z0-9+/]{2}==|[A-Za-z0-9+/]{3}=|[A-Za-z0-9+/]{4})$/.test(
		value,
	);
```

**TypeScript version**

```js
const isBase64 = (value: string): boolean =>
	/^(?:[A-Za-z0-9+/]{4})*(?:[A-Za-z0-9+/]{2}==|[A-Za-z0-9+/]{3}=|[A-Za-z0-9+/]{4})$/.test(
		value,
	);
```
