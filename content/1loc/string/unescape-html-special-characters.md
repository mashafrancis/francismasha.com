title: Unescape HTML special characters
category: String
---

**JavaScript version**

```js
const unescape = (str) =>
	str
		.replace(/&amp;/g, '&')
		.replace(/&lt;/g, '<')
		.replace(/&gt;/g, '>')
		.replace(/&#0*39;/g, "'")
		.replace(/&quot;/g, '"');
```

**TypeScript version**

```js
const unescape = (str: string): string =>
	str
		.replace(/&amp;/g, '&')
		.replace(/&lt;/g, '<')
		.replace(/&gt;/g, '>')
		.replace(/&#0*39;/g, "'")
		.replace(/&quot;/g, '"');
```
