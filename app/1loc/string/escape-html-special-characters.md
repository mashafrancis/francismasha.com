---

**JavaScript version**

```js
const escape = (str) =>
	str
		.replace(/&/g, '&amp;')
		.replace(/</g, '&lt;')
		.replace(/>/g, '&gt;')
		.replace(/'/g, '&#39;')
		.replace(/"/g, '&quot;');

// Or
const escape = (str) =>
	str.replace(
		/[&<>"']/g,
		(m) =>
			({ '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#39;' }[
				m
			]),
	);
```
